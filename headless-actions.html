🔌 API Integration Examples

Trigger Headless Action via API

// POST /api/v1/headless-actions/trigger
const response = await fetch('https://api.mcp-health.io/v1/headless-actions/trigger', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    action: 'auto-backfill',
    pipeline: 'google-ads-apac',
    params: {
      startDate: '2026-06-14',
      endDate: '2026-06-17',
      priority: 'high'
    }
  })
});

const result = await response.json();
// { jobId: "bf-123456", status: "queued", estimatedDuration: "2h" }
