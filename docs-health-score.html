1. Simple Health Score (Basic Approach)

def calculate_simple_health_score(pipelines):

    """


    Simple health score based on pipeline status distribution


    Returns: 0-100 score


    """


    if not pipelines:


        return 0

    

    total = len(pipelines)


    healthy = sum(1 for p in pipelines if p.status == 'healthy')


    warning = sum(1 for p in pipelines if p.status == 'warning')


    critical = sum(1 for p in pipelines if p.status == 'critical')

    

    # Weight: Healthy=100%, Warning=50%, Critical=0%


    score = ((healthy * 100) + (warning * 50) + (critical * 0)) / total

    

    return round(score)

2. Weighted Health Score (Recommended)

from datetime import datetime, timedelta

from typing import List, Dict



def calculate_weighted_health_score(pipelines: List[Dict]) -> Dict:


    """


    Weighted health score considering multiple factors

    

    Factors:


    - Pipeline status (40% weight)


    - Success rate (30% weight)


    - Data freshness (15% weight)


    - Token health (10% weight)


    - Trend direction (5% weight)

    

    Returns: {


        'score': 0-100,


        'grade': 'Excellent'|'Good'|'Fair'|'Poor'|'Critical',


        'breakdown': {...}


    }


    """


    if not pipelines:


        return {'score': 0, 'grade': 'Critical', 'breakdown': {}}

    

    total_pipelines = len(pipelines)

    

    # 1. STATUS SCORE (40% weight)


    status_counts = {


        'healthy': sum(1 for p in pipelines if p.get('status') == 'healthy'),


        'warning': sum(1 for p in pipelines if p.get('status') == 'warning'),


        'critical': sum(1 for p in pipelines if p.get('status') == 'critical')


    }

    

    status_score = (


        (status_counts['healthy'] * 100) +


        (status_counts['warning'] * 50) +


        (status_counts['critical'] * 0)


    ) / total_pipelines

    

    # 2. SUCCESS RATE SCORE (30% weight)


    success_rates = [p.get('success_rate', 0) for p in pipelines]


    avg_success_rate = sum(success_rates) / len(success_rates) if success_rates else 0


    success_score = avg_success_rate

    

    # 3. DATA FRESHNESS SCORE (15% weight)


    now = datetime.now()


    freshness_scores = []

    

    for pipeline in pipelines:


        last_run_str = pipeline.get('last_run', '1 hour ago')

        

        # Parse last_run string


        if 'min' in last_run_str:


            minutes = int(last_run_str.split()[0])


            freshness = 100 if minutes < 30 else 90


        elif 'hour' in last_run_str:


            hours = int(last_run_str.split()[0])


            if hours < 2:


                freshness = 85


            elif hours < 6:


                freshness = 70


            elif hours < 24:


                freshness = 50


            else:


                freshness = 20


        elif 'day' in last_run_str:


            days = int(last_run_str.split()[0])


            freshness = max(0, 100 - (days * 30))


        else:


            freshness = 100

        

        freshness_scores.append(freshness)

    

    freshness_score = sum(freshness_scores) / len(freshness_scores)

    

    # 4. TOKEN HEALTH SCORE (10% weight)


    token_issues = sum(1 for p in pipelines if p.get('needs_auth') or 


                       (p.get('error') and 'token' in p.get('error', '').lower()))


    token_healthy = total_pipelines - token_issues


    token_score = (token_healthy / total_pipelines) * 100

    

    # 5. TREND SCORE (5% weight)


    trend_scores = []


    for pipeline in pipelines:


        trend = pipeline.get('trend', [])


        if len(trend) >= 2:


            # Compare recent average to older average


            recent_avg = sum(trend[-7:]) / len(trend[-7:]) if len(trend) >= 7 else sum(trend) / len(trend)


            older_avg = sum(trend[:7]) / 7 if len(trend) >= 14 else recent_avg

            

            if recent_avg > older_avg:


                trend_score = 100  # Improving


            elif recent_avg == older_avg:


                trend_score = 75   # Stable


            else:


                trend_score = 50   # Declining


        else:


            trend_score = 75  # No trend data, assume stable

        

        trend_scores.append(trend_score)

    

    avg_trend_score = sum(trend_scores) / len(trend_scores)

    

    # CALCULATE WEIGHTED TOTAL


    total_score = (


        status_score * 0.40 +


        success_score * 0.30 +


        freshness_score * 0.15 +


        token_score * 0.10 +


        avg_trend_score * 0.05


    )

    

    # DETERMINE GRADE


    if total_score >= 90:


        grade = 'Excellent'


    elif total_score >= 80:


        grade = 'Good'


    elif total_score >= 70:


        grade = 'Fair'


    elif total_score >= 50:


        grade = 'Poor'


    else:


        grade = 'Critical'

    

    return {


        'score': round(total_score),


        'grade': grade,


        'breakdown': {


            'status': {


                'score': round(status_score),


                'weight': '40%',


                'healthy': status_counts['healthy'],


                'warning': status_counts['warning'],


                'critical': status_counts['critical']


            },


            'success_rate': {


                'score': round(success_score, 1),


                'weight': '30%',


                'average': f"{round(avg_success_rate, 1)}%"


            },


            'freshness': {


                'score': round(freshness_score),


                'weight': '15%'


            },


            'token_health': {


                'score': round(token_score),


                'weight': '10%',


                'issues': token_issues


            },


            'trend': {


                'score': round(avg_trend_score),


                'weight': '5%'


            }


        }


    }

3. Real-Time Health Score (Database Query)

-- SQL query to calculate health score in real-time

WITH pipeline_metrics AS (


    SELECT 


        p.id,


        p.status,


        COALESCE(m.success_rate, 0) as success_rate,


        EXTRACT(EPOCH FROM (NOW() - pr.completed_at)) / 3600 as hours_since_last_run,


        CASE 


            WHEN c.token_expires_at < NOW() THEN 1


            WHEN c.token_expires_at < NOW() + INTERVAL '2 days' THEN 0.5


            ELSE 0


        END as token_issue_score


    FROM pipelines p


    LEFT JOIN pipeline_health_metrics m ON p.id = m.pipeline_id 


        AND m.metric_date = CURRENT_DATE


    LEFT JOIN pipeline_runs pr ON p.id = pr.pipeline_id


        AND pr.id = (


            SELECT id FROM pipeline_runs 


            WHERE pipeline_id = p.id 


            ORDER BY completed_at DESC 


            LIMIT 1


        )


    LEFT JOIN pipeline_credentials c ON p.id = c.pipeline_id 


        AND c.is_active = true


)


SELECT


    -- Status score (40%)


    ROUND(


        (SUM(CASE WHEN status = 'healthy' THEN 100 WHEN status = 'warning' THEN 50 ELSE 0 END) 


         / COUNT(*)) * 0.40


    ) as status_component,

    

    -- Success rate score (30%)


    ROUND(AVG(success_rate) * 0.30, 2) as success_component,

    

    -- Freshness score (15%)


    ROUND(


        (SUM(CASE 


            WHEN hours_since_last_run < 1 THEN 100


            WHEN hours_since_last_run < 6 THEN 85


            WHEN hours_since_last_run < 24 THEN 70


            WHEN hours_since_last_run < 72 THEN 40


            ELSE 0


        END) / COUNT(*)) * 0.15


    ) as freshness_component,

    

    -- Token health score (10%)


    ROUND(


        ((COUNT(*) - SUM(token_issue_score)) / COUNT(*) * 100) * 0.10


    ) as token_component,

    

    -- Total score


    ROUND(


        (SUM(CASE WHEN status = 'healthy' THEN 100 WHEN status = 'warning' THEN 50 ELSE 0 END) 


         / COUNT(*)) * 0.40 +


        AVG(success_rate) * 0.30 +


        (SUM(CASE 


            WHEN hours_since_last_run < 1 THEN 100


            WHEN hours_since_last_run < 6 THEN 85


            WHEN hours_since_last_run < 24 THEN 70


            WHEN hours_since_last_run < 72 THEN 40


            ELSE 0


        END) / COUNT(*)) * 0.15 +


        ((COUNT(*) - SUM(token_issue_score)) / COUNT(*) * 100) * 0.10


    ) as total_health_score,

    

    -- Supporting stats


    COUNT(*) as total_pipelines,


    SUM(CASE WHEN status = 'healthy' THEN 1 ELSE 0 END) as healthy_count,


    SUM(CASE WHEN status = 'warning' THEN 1 ELSE 0 END) as warning_count,


    SUM(CASE WHEN status = 'critical' THEN 1 ELSE 0 END) as critical_count


FROM pipeline_metrics;

4. Advanced Health Score with Severity Weights

def calculate_advanced_health_score(pipelines: List[Dict]) -> Dict:

    """


    Advanced scoring with severity-weighted penalties

    

    Penalties:


    - Token expired: -15 points


    - Pipeline down >24h: -20 points


    - Pipeline down >72h: -30 points


    - Rate limit issues: -5 points per occurrence


    - Data quality issues: -10 points


    """


    if not pipelines:


        return {'score': 0, 'grade': 'Critical'}

    

    # Start with base score


    base_score = calculate_weighted_health_score(pipelines)['score']

    

    # Apply penalties


    penalty = 0

    

    for pipeline in pipelines:


        error = pipeline.get('error', '').lower()


        last_run = pipeline.get('last_run', '')

        

        # Token expiry penalty


        if pipeline.get('needs_auth') or 'token expired' in error:


            penalty += 15

        

        # Downtime penalties


        if 'day' in last_run:


            days = int(last_run.split()[0])


            if days >= 3:


                penalty += 30


            elif days >= 1:


                penalty += 20

        

        # Rate limit penalty


        if 'rate limit' in error:


            penalty += 5

        

        # API deprecation penalty (high severity)


        if 'deprecated' in error:


            penalty += 10

    

    # Apply penalty cap (max -40 points)


    penalty = min(penalty, 40)

    

    final_score = max(0, base_score - penalty)

    

    # Determine grade


    if final_score >= 90:


        grade = 'Excellent'


        status = 'Very High'


        color = '#22543d'  # Green


    elif final_score >= 80:


        grade = 'Good'


        status = 'High'


        color = '#2d5016'  # Light green


    elif final_score >= 70:


        grade = 'Fair'


        status = 'Moderate'


        color = '#ecb22e'  # Yellow


    elif final_score >= 50:


        grade = 'Poor'


        status = 'Low'


        color = '#ed8936'  # Orange


    else:


        grade = 'Critical'


        status = 'Very Low'


        color = '#e01e5a'  # Red

    

    return {


        'score': round(final_score),


        'grade': grade,


        'status': status,


        'color': color,


        'base_score': base_score,


        'penalty': penalty,


        'recommendation': get_recommendation(final_score, pipelines)


    }



def get_recommendation(score: float, pipelines: List[Dict]) -> str:


    """Generate actionable recommendations based on score"""


    critical_count = sum(1 for p in pipelines if p.get('status') == 'critical')


    token_issues = sum(1 for p in pipelines if p.get('needs_auth'))

    

    if score >= 90:


        return "All systems operating normally. Continue monitoring."


    elif score >= 80:


        return "Overall health is good. Address minor warnings proactively."


    elif score >= 70:


        if token_issues > 0:


            return f"Re-authenticate {token_issues} pipeline(s) immediately to restore health."


        else:


            return f"Investigate and resolve {critical_count} critical pipeline(s) to improve health."


    else:


        actions = []


        if token_issues > 0:


            actions.append(f"re-authenticate {token_issues} pipeline(s)")


        if critical_count > 0:


            actions.append(f"repair {critical_count} failed pipeline(s)")

        

        return f"Urgent action required: {', '.join(actions)}. Estimated recovery time: {critical_count * 20} minutes."
